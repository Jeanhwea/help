  from string import Template
  s = Template('$friut is $color') # using $$ to diplay $
  d = {'friut': 'apple', 'color': 'red'}
  s.substitute(friut='banana', color='yellow') #>>> banana is yellow
  s.substitute(d) #>>> apple is red
